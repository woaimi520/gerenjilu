		        mVp.setAdapter(new FragmentPagerAdapter(getSupportFragmentManager()) {
            @Override
            public Fragment getItem(int position) {
                GuideFragment fragment = new GuideFragment();//这里为每个item设置一个fragment
                Bundle bundle = new Bundle();
                bundle.putInt("guide", position);
                fragment.setArguments(bundle);
                return fragment;
            }

            @Override
            public int getCount() {
                return Constant.GUIDE_TEXT_TITLE_SJ.length;
            }


        });

		 @Override
    public void onViewCreated(@NonNull View view, @Nullable Bundle savedInstanceState) {
        super.onViewCreated(view, savedInstanceState);

        int guide_index = 0;
        Bundle bundle = getArguments();
        if(bundle != null){
            guide_index = bundle.getInt("guide"); //这里取位置
        }
        switch (guide_index) {
            case 0:
                Drawable drawable0 = ContextCompat.getDrawable(getContext(), R.mipmap.guild_image_0);
                mImage.setImageDrawable(drawable0);
                break;
            case 1:
                Drawable drawable1 = ContextCompat.getDrawable(getContext(), R.mipmap.guild_image_01);
                mImage.setImageDrawable(drawable1);
                break;
            case 2:
                Drawable drawable2 = ContextCompat.getDrawable(getContext(), R.mipmap.guild_image_02);
                mImage.setImageDrawable(drawable2);
                break;
        }
    }
