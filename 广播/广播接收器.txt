	注册
	RemindReceiver2 dynamicReceiver;
	

    @Override
    public void onActivityCreated(@Nullable Bundle savedInstanceState) {
		IntentFilter filter = new IntentFilter();
        filter.addAction(RemindReceiver2.ACTION);//动态注册用action 我用组件名 收不到广播
        dynamicReceiver = new RemindReceiver2();
        getContext().registerReceiver(dynamicReceiver,filter);
    }
	
	去注册
	    public void onDetach() {
       
        getContext().unregisterReceiver(dynamicReceiver);

    }
